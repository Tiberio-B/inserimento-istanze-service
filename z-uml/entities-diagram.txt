package "Model" as mod #Motivation {

		abstract class Istanza {
			- qualitaRichiedente*** : QualitaRichiedente
			- categoriaDeposito : Categoria
			- stato : Stato
			- pagamentoASoggettiDeposito : boolean
			- modalitaPagamento : modalitaPagamento
			- importoRichiesto : BigDecimal
		}

		class Soggetto {
			- nome : String
			- cognome : String
			- cf : String
			- sesso : Sesso
			- nascita : Indirizzo
			- partitaIva : String
			- irreperibile : boolean
			- relataNotificaSNR : Documento
		}

		class Deposito {
			- causale* : String
			- importo* : BigDecimal
		}

		class Richiesta {
			- numero** : Long 
			- data* : LocalDateTime 
		}
		class Protocollo {
			- numero** : Long
			- data* : LocalDateTime 
		}
		class Rts {
			- id** : Long
		}
		
		class Opera {
			- descrizione* : Opera
			- regione*** : Regione
		}

		class Bene {
			datiCatastali* : Catasto
			proprietariCatastali* : Set<Soggetto>
		}
		
}

'########## Relazioni ##########
